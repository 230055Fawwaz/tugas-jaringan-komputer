! ==========================================================
! PROYEK MINI JARINGAN KOMPUTER - MANAGEMENTSWITCH (L2 SWITCH)
! ==========================================================

enable
configure terminal
hostname ManagementSwitch

! ----------------------------------------------------------
! 1. VLAN DATABASE CONFIGURATION
! ----------------------------------------------------------
vlan 30
 name MANAGEMENT_AREA
exit

! ----------------------------------------------------------
! 2. LAYER 2 PORT CONFIGURATION
! ----------------------------------------------------------
! Port ke arah CoreSwitch (Gi0/2)
interface GigabitEthernet0/0
 switchport mode access
 switchport access vlan 30
 spanning-tree portfast
 no shutdown
exit

! Port ke arah ManagementServer
interface GigabitEthernet0/1
 switchport mode access
 switchport access vlan 30
 spanning-tree portfast
 no shutdown
exit

! Port ke arah NetAdminPC
interface GigabitEthernet0/2
 switchport mode access
 switchport access vlan 30
 spanning-tree portfast
 no shutdown
exit

! ----------------------------------------------------------
! 3. SWITCH MANAGEMENT CONFIGURATION (SVI)
! ----------------------------------------------------------
interface Vlan30
 ip address 10.10.30.2 255.255.255.0
 no shutdown
exit

ip default-gateway 10.10.30.1

! ----------------------------------------------------------
! 4. SAVE CONFIGURATION
! ----------------------------------------------------------
exit
copy running-config startup-config
