enable
configure terminal
hostname Access-Switch-2

! Daftarkan VLAN
vlan 20
name Admin
vlan 50
name CCTV
exit

! 2. Konfigurasi Port Trunk yang mengarah ke Core Switch
interface gigabitEthernet 1/0/1
switchport trunk encapsulation dot1q
switchport mode trunk
description Jalur_ke_CoreSwitch
exit

! 3. Alokasikan Port untuk 2 PC Admin
interface range gigabitEthernet 1/0/2 - 3
switchport mode access
switchport access vlan 20
description PC_Admin
exit

! 4. Alokasikan Port untuk 4 CCTV IP
interface range gigabitEthernet 1/0/4 - 7
switchport mode access
switchport access vlan 50
description Kamera_CCTV
exit

! Mematikan port rentang yang tidak digunakan
interface range gigabitEthernet 1/0/8 - 24
shutdown
exit

interface range gigabitEthernet 1/0/2 - 7
spanning-tree portfast
exit

exit
copy running-config startup-config
