enable
configure terminal
hostname CoreSwitch

! Membuat Database VLAN
vlan 10
name POS_Kasir
vlan 20
name Admin
vlan 30
name Server
vlan 40
name IoT_Inventaris
vlan 50
name CCTV
vlan 60
name WiFi_Staff
vlan 70
name WiFi_Pelanggan
exit

! Konfigurasi Port Trunk ke Router
interface gigabitEthernet 1/0/1
switchport trunk encapsulation dot1q
switchport mode trunk
description Jalur_ke_EdgeRouter
exit

! Jalur Trunk ke Access-Switch-1 (VLAN 10)
interface gigabitEthernet 1/0/2
switchport trunk encapsulation dot1q
switchport mode trunk
description Jalur_ke_AccessSwitch1
exit

! Jalur Trunk ke Access-Switch-2 (VLAN 50)
interface gigabitEthernet 1/0/3
switchport trunk encapsulation dot1q
switchport mode trunk
description Jalur_ke_AccessSwitch2
exit

! 3. Konfigurasi Port Khusus untuk Server Lokal (VLAN 30)
interface gigabitEthernet 1/0/4
switchport mode access
switchport access vlan 30
description Koneksi_Server_Lokal
exit

! Port untuk AP IoT (VLAN 40)
interface gigabitEthernet 1/0/6
switchport mode access
switchport access vlan 40
description AP_IoT_Inventaris
exit

! Port untuk AP WiFi Staff (VLAN 60)
interface gigabitEthernet 1/0/5
switchport mode access
switchport access vlan 60
description AP_WiFi_Staff
exit

! Port untuk AP WiFi Pelanggan (VLAN 70)
interface gigabitEthernet 1/0/7
switchport mode access
switchport access vlan 70
description AP_WiFi_Pelanggan
exit

! Mematikan port rentang yang tidak digunakan (contoh port 8 sampai 24)
interface range gigabitEthernet 1/0/8 - 24
shutdown
exit

! Mengaktifkan PortFast pada port access agar PC/Server cepat mendapat IP DHCP
interface range gigabitEthernet 1/0/4 , gigabitEthernet 1/0/5 , gigabitEthernet 1/0/6 , gigabitEthernet 1/0/7
spanning-tree portfast
exit

exit
copy running-config startup-config
